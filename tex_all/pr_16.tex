\textbf{Задача 1.} Представьте в виде корня: $5^{1/2}$\par\vspace{10pt}
\textbf{Задача 2.} Представьте в виде степени: $\sqrt[3]{7}$\par\vspace{10pt}
\textbf{Задача 3.} Вычислите: $8^{\frac{1}{3}}$\par\vspace{10pt}
\textbf{Задача 4.} Вычислите: $16^{\frac{1}{2}}$\par\vspace{10pt}
\textbf{Задача 5.} Упростите: $a^{\frac{1}{2}} \cdot a^{\frac{1}{3}}$\par\vspace{10pt}
\textbf{Задача 6.} Упростите: $\left(x^{\frac{1}{2}}\right)^4$\par\vspace{10pt}
\textbf{Задача 7.} Вычислите: $27^{\frac{2}{3}}$\par\vspace{10pt}
\textbf{Задача 8.} Упростите выражение: $\dfrac{a^{\frac{3}{4}} \cdot a^{\frac{1}{4}}}{a^{\frac{1}{2}}}$\par\vspace{10pt}
\textbf{Задача 9.} Вычислите: $16^{\frac{3}{4}}$\par\vspace{10pt}
\textbf{Задача 10.} Объём бетонного куба $V = a^3$. Выразите сторону $a$ через $V$ с помощью степени с рациональным показателем и найдите $a$, если $V = 125$ м³.\par\vspace{10pt}
\newpage
\section*{Ответы}

\begin{enumerate}
  \item Ответ: $\sqrt{5}$
  \item Ответ: $7^{\frac{1}{3}}$
  \item Ответ: 2.
  \item Ответ: 4.
  \item Ответ: $a^{\frac{5}{6}}$
  \item Ответ: $x^2$
  \item Ответ: 9.
  \item Ответ: $a^{\frac{1}{2}}$ или $\sqrt{a}$
  \item Ответ: 8.
  \item Ответ: $a = V^{\frac{1}{3}};\ a = 5$ м.
\end{enumerate}
