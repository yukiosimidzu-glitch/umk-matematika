\textbf{Задача 1.} Переведите 30$^\circ$ в радианы.\par\vspace{10pt}
\textbf{Задача 2.} Переведите 90$^\circ$ в радианы.\par\vspace{10pt}
\textbf{Задача 3.} Переведите $\dfrac{\pi}{4}$ радиан в градусы.\par\vspace{10pt}
\textbf{Задача 4.} Переведите 180$^\circ$ в радианы.\par\vspace{10pt}
\textbf{Задача 5.} Переведите 150$^\circ$ в радианы.\par\vspace{10pt}
\textbf{Задача 6.} Переведите $\dfrac{2\pi}{3}$ радиан в градусы.\par\vspace{10pt}
\textbf{Задача 7.} Заполните пропуск: 270$^\circ$ = \_\_\_ рад.\par\vspace{10pt}
\textbf{Задача 8.} Переведите 315$^\circ$ в радианы. Ответ дайте в виде дроби.\par\vspace{10pt}
\textbf{Задача 9.} Угол наклона кровли 40$^\circ$. Выразите в радианах (округлите до сотых, $\pi \approx 3{,}14$).\par\vspace{10pt}
\textbf{Задача 10.} Кран повернулся на $\dfrac{5\pi}{6}$ радиан. Сколько это в градусах?\par\vspace{10pt}
\newpage
\section*{Ответы}

\begin{enumerate}
  \item Ответ: $\dfrac{\pi}{6}$
  \item Ответ: $\dfrac{\pi}{2}$
  \item Ответ: 45$^\circ$.
  \item Ответ: $\pi$
  \item Ответ: $\dfrac{5\pi}{6}$
  \item Ответ: 120$^\circ$.
  \item Ответ: $\dfrac{3\pi}{2}$
  \item Ответ: $\dfrac{7\pi}{4}$
  \item Ответ: $\approx$ 0,70 рад.
  \item Ответ: 150$^\circ$.
\end{enumerate}
