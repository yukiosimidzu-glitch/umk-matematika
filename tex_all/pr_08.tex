\textbf{Задача 1.} Найдите производную: $f(x) = 5x^3$\par\vspace{10pt}
\textbf{Задача 2.} Найдите производную: $f(x) = x^4 + 2x^3 - x$\par\vspace{10pt}
\textbf{Задача 3.} Найдите производную: $f(x) = 7$\par\vspace{10pt}
\textbf{Задача 4.} Найдите $f'(2)$, если $f(x) = x^2 + 3x$\par\vspace{10pt}
\textbf{Задача 5.} Найдите производную: $f(x) = (x + 1)(x - 4)$\par\vspace{10pt}
\textbf{Задача 6.} Найдите производную: $f(x) = (2x + 1)(x^2 - 3)$\par\vspace{10pt}
\textbf{Задача 7.} Решите уравнение $f'(x) = 0$, если $f(x) = x^3 - 12x + 5$\par\vspace{10pt}
\textbf{Задача 8.} Найдите производную и вычислите $f'(-1)$: $f(x) = (x^2 + 2)(x - 5)$\par\vspace{10pt}
\textbf{Задача 9.} Объём бетонной смеси при твердении меняется по закону $V(t) = 2t^2 + 4t$ (t — часы). Найдите скорость изменения объёма через 3 часа.\par\vspace{10pt}
\textbf{Задача 10.} Прочность строительной конструкции зависит от толщины $x$ по формуле $P(x) = -x^2 + 12x$. При какой толщине прочность максимальна?\par\vspace{10pt}
\newpage
\section*{Ответы}

\begin{enumerate}
  \item Ответ: $f'(x) = 15x^2$
  \item Ответ: $f'(x) = 4x^3 + 6x^2 - 1$
  \item Ответ: $f'(x) = 0$
  \item Ответ: $f'(2) = 7$
  \item Ответ: $f'(x) = 2x - 3$
  \item Ответ: $f'(x) = 6x^2 + 2x - 6$
  \item Ответ: $x = -2;\ x = 2$
  \item Ответ: $f'(x) = 3x^2 - 10x + 2;\ f'(-1) = 15$
  \item Ответ: 16.
  \item Ответ: толщина 6 единиц, $P\_{\max} = 36$.
\end{enumerate}
