\textbf{Задача 1.} Решите уравнение: $2^x = 8$\par\vspace{10pt}
\textbf{Задача 2.} Решите уравнение: $5^x = 25$\par\vspace{10pt}
\textbf{Задача 3.} Решите уравнение: $3^x = 1$\par\vspace{10pt}
\textbf{Задача 4.} Решите уравнение: $10^x = 1000$\par\vspace{10pt}
\textbf{Задача 5.} Решите уравнение: $5^{2x-1} = 125$\par\vspace{10pt}
\textbf{Задача 6.} Решите уравнение: $3^{x+2} + 3^x = 90$\par\vspace{10pt}
\textbf{Задача 7.} Решите уравнение: $2^{2x} - 6 \cdot 2^x + 8 = 0$\par\vspace{10pt}
\textbf{Задача 8.} Решите уравнение: $4^x - 3 \cdot 2^{x+1} + 8 = 0$\par\vspace{10pt}
\textbf{Задача 9.} Решите уравнение: $9^x - 4 \cdot 3^x + 3 = 0$\par\vspace{10pt}
\textbf{Задача 10.} Население города растёт по закону $N(t) = N\_0 \cdot 2^{\frac{t}{10}}$, где t — годы. Через сколько лет население удвоится?\par\vspace{10pt}
\newpage
\section*{Ответы}

\begin{enumerate}
  \item Ответ: $x = 3$
  \item Ответ: $x = 2$
  \item Ответ: $x = 0$
  \item Ответ: $x = 3$
  \item Ответ: $x = 2$
  \item Ответ: $x = 2$
  \item Ответ: $x = 1;\ x = 2$
  \item Ответ: $x = 1;\ x = 2$
  \item Ответ: $x = 0;\ x = 1$
  \item Ответ: через 10 лет.
\end{enumerate}
