\textbf{Задача 1.} Решите неравенство: $\log\_{2} x > 3$\par\vspace{10pt}
\textbf{Задача 2.} Решите неравенство: $\log\_{3} (x - 1) \le 2$\par\vspace{10pt}
\textbf{Задача 3.} Решите неравенство: $\log\_{\frac{1}{2}} x < 3$\par\vspace{10pt}
\textbf{Задача 4.} Решите неравенство: $\log\_{5} (2x + 1) \ge 0$\par\vspace{10pt}
\textbf{Задача 5.} Решите неравенство: $\log\_{2} (x + 3) + \log\_{2} (x - 3) \ge 4$\par\vspace{10pt}
\textbf{Задача 6.} Решите неравенство: $\log\_{3} (x^2 - 4) - \log\_{3} (x + 2) < 1$\par\vspace{10pt}
\textbf{Задача 7.} Решите неравенство: $\log\_{0{,}5}^{2} x - \log\_{0{,}5} x - 2 \le 0$\par\vspace{10pt}
\textbf{Задача 8.} Решите неравенство: $\log\_{2} (3x - 5) < \log\_{2} (x + 3)$\par\vspace{10pt}
\textbf{Задача 9.} Решите неравенство: $\log\_{3} (x^2 - 2x) \le 1$\par\vspace{10pt}
\textbf{Задача 10.} Решите неравенство: $\frac{\log\_{2} x - 3}{\log\_{2} x - 1} \ge 1$\par\vspace{10pt}
